 27:       $this->api = $bag->api;

 29:     }

 39:     //echo __METHOD__.",".__LINE__."\n";//##DEBUG
 40:     if ($ev->isCancelled()) return;

 42:     if (!$this->getState("fz",$p,false)) return;
